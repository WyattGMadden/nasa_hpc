\newcommand{\practica}[3]{%
  node (p#1) [practica] {%
    \scriptsize
    \renewcommand{\arraystretch}{1.3}%
    \begin{tabularx}{\linewidth}{@{}l>{\raggedright\arraybackslash}X@{}}
      \textbf{Task:}   & #2 \\

    \end{tabularx}%
  }%
}

% Draw background
\newcommand{\background}[5]{%
  \begin{pgfonlayer}{background}
    % compute the two corners of the box
    \path (#1.west |- #2.north)+(-1.1,0.7) coordinate (a1);
    \path (#3.east |- #4.south)+(1.1,-0.7)    coordinate (a2);
    % draw the dashed background
    \path[fill=green!15,rounded corners,draw=black!50,dashed]
      (a1) rectangle (a2);
    % place the label exactly at the horizontal midpoint,
    % just 2pt inside the top edge
    \node[anchor=center,yshift=-9pt] at
      ($ (a1)!0.5!(a2|-a1) $)
      {\scriptsize\itshape #5};
  \end{pgfonlayer}%
}
